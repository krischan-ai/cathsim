### 4.3 WebSocket 客户端（GDScript）

```gdscript
# websocket_client.gd
extends Node

var socket := WebSocketPeer.new()
var server_url := "ws://localhost:8000/ws/session"
var connected := false
var session_id := ""

signal connected_to_server
signal connection_closed
signal state_received(state: Dictionary)
signal path_received(path: Dictionary)

func _ready():
    connect_to_server()

func connect_to_server():
    var err = socket.connect_to_url(server_url)
    if err != OK:
        print("WebSocket connection failed: ", err)
        return

func _process(delta):
    socket.poll()
    var state = socket.get_ready_state()
    
    if state == WebSocketPeer.STATE_OPEN and not connected:
        connected = true
        connected_to_server.emit()
        # 发送 session_start
        send_message("session_start", {
            "phantom": "low_tort",
            "target": "bca"
        })
    
    while socket.get_available_packet_count() > 0:
        var packet = socket.get_packet().get_string_from_utf8()
        var msg = JSON.parse_string(packet)
        handle_message(msg)
    
    if state == WebSocketPeer.STATE_CLOSED and connected:
        connected = false
        connection_closed.emit()

func send_message(type_name: String, data: Dictionary):
    var msg = {
        "type": type_name,
        "session_id": session_id,
        "timestamp": int(Time.get_unix_time_from_system() * 1000),
        "data": data
    }
    socket.send_text(JSON.stringify(msg))

func handle_message(msg: Dictionary):
    match msg.get("type"):
        "session_started":
            session_id = msg.get("data", {}).get("session_id", "")
        "state_update":
            state_received.emit(msg.get("data", {}))
        "path_response":
            path_received.emit(msg.get("data", {}))
        "error":
            push_error("Server error: ", msg.get("data", {}))
        "ping":
            send_message("pong", {})

func send_control(delta_push: float, delta_rotate: float):
    if not connected:
        return
    send_message("control", {
        "delta_push": delta_push,
        "delta_rotate": delta_rotate
    })
```

### 4.4 手柄输入映射

```gdscript
# input_handler.gd
extends Node

signal control_received(delta_push: float, delta_rotate: float)

@export var deadzone := 0.15
@export var max_push_speed := 1.0
@export var max_rotate_speed := 0.5

func _input(event):
    # Xbox 手柄映射
    if event is InputEventJoypadMotion:
        match event.axis:
            JOY_AXIS_LEFT_Y:  # 左摇杆 Y → 推进
                var push = -event.axis_value  # 上推=前进
                if abs(push) < deadzone:
                    push = 0.0
                push = clamp(push, -1.0, 1.0)
            JOY_AXIS_RIGHT_X:  # 右摇杆 X → 旋转
                var rotate = event.axis_value
                if abs(rotate) < deadzone:
                    rotate = 0.0
                rotate = clamp(rotate, -1.0, 1.0)
                control_received.emit(push, rotate)
    
    # 键盘备选
    if event is InputEventKey:
        var push := 0.0
        var rotate := 0.0
        if Input.is_key_pressed(KEY_W): push -= 1.0
        if Input.is_key_pressed(KEY_S): push += 1.0
        if Input.is_key_pressed(KEY_A): rotate -= 1.0
        if Input.is_key_pressed(KEY_D): rotate += 1.0
        if push != 0.0 or rotate != 0.0:
            control_received.emit(push, rotate * 0.5)
```

## 六、Docker 部署

### 6.1 Dockerfile

```dockerfile
FROM python:3.10-slim

WORKDIR /app

# 系统依赖
RUN apt-get update && apt-get install -y \
    libgl1-mesa-glx libglib2.0-0 libglfw3 \
    && rm -rf /var/lib/apt/lists/*

# Python 依赖
COPY services/requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 源码
COPY src/ ./src/
COPY services/ ./services/
COPY data/ ./data/

# 环境变量
ENV MUJOCO_GL=egl
ENV PYTHONPATH=/app

EXPOSE 8000

CMD ["uvicorn", "services.main:app", "--host", "0.0.0.0", "--port", "8000"]
```

### 6.2 docker-compose.yml

```yaml
version: '3.8'

services:
  cathsim-server:
    build: .
    ports:
      - "8000:8000"
    environment:
      - MUJOCO_GL=egl
      - PYTHONPATH=/app
    volumes:
      - ./data:/app/data
    deploy:
      resources:
        limits:
          memory: 4G

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data

  # Godot 客户端需要本地 GPU 渲染，不部署在 Docker 中

volumes:
  redis-data:
```

## 九、常见问题

### Q1: GLFW 窗口无法打开（无头服务器）

```bash
# 使用 EGL 渲染（无头）
$env:MUJOCO_GL="egl"
# 或在 Docker 中使用 osmesa
$env:MUJOCO_GL="osmesa"
```

### Q2: WebSocket 连接失败

```bash
# 检查服务是否运行
curl http://localhost:8000/api/v1/health

# 检查端口占用
netstat -ano | findstr :8000
```

### Q3: Godot GLB 模型不显示

```bash
# 确保 GLB 文件导出时包含材质和动画
# 在 3D Slicer 中导出的 VTK 需先转为 GLB
# 使用 trimesh:
python -c "
import trimesh
mesh = trimesh.load('data/vpp_assets/blood_vessels.vtk')
mesh.export('data/vpp_assets/blood_vessels.glb')
"
```
